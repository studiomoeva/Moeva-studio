  <script>
    window.addEventListener('scroll', () => {
      const nav = document.getElementById('navbar');
      if(window.scrollY > 40){ nav.classList.add('scrolled'); } else { nav.classList.remove('scrolled'); }
    });

    const hamburger = document.getElementById('hamburger');
    const mobileNav = document.getElementById('mobileNav');
    hamburger.addEventListener('click', () => { mobileNav.classList.toggle('open'); });
    function closeMobileNav(){ mobileNav.classList.remove('open'); }

    function scrollTo(selector){
      const el = document.querySelector(selector);
      if(el){ el.scrollIntoView({behavior:'smooth'}); }
    }

    const observer = new IntersectionObserver((entries)=>{
      entries.forEach(entry => { if(entry.isIntersecting){ entry.target.classList.add('visible'); } });
    },{threshold:0.12});
    document.querySelectorAll('.reveal').forEach(el => observer.observe(el));

    function toggleFaq(item){
      const isOpen = item.classList.contains('open');
      document.querySelectorAll('.faq-item').forEach(el => el.classList.remove('open'));
      if(!isOpen){ item.classList.add('open'); }
    }

    function switchTab(btn, paneId){
      document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
      document.querySelectorAll('.tab-pane').forEach(p => p.classList.remove('active'));
      btn.classList.add('active');
      document.getElementById(paneId).classList.add('active');
    }

    function filterSch(btn, tipo){
      document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
      btn.classList.add('active');
      document.querySelectorAll('.sch-clase').forEach(cl => {
        if(tipo === 'all' || cl.getAttribute('data-tipo') === tipo){
          cl.style.display = 'grid';
        } else {
          cl.style.display = 'none';
        }
      });
    }

    let selectedClaseEl = null;
    function selClase(el, status){
      if(status === 'full'){
        document.querySelectorAll('.sch-clase').forEach(c => c.classList.remove('selected'));
        el.classList.add('selected');
        selectedClaseEl = el;
        document.getElementById('btnReservar').disabled = true;
        document.getElementById('btnReservar').style.display = 'none';
        document.getElementById('waitlistBtn').classList.add('show');
        showToast('Clase llena', 'Puedes unirte a la lista de espera automática.');
        return;
      }
      document.querySelectorAll('.sch-clase').forEach(c => c.classList.remove('selected'));
      el.classList.add('selected');
      selectedClaseEl = el;
      document.getElementById('waitlistBtn').classList.remove('show');
      document.getElementById('btnReservar').style.display = 'block';
      document.getElementById('btnReservar').disabled = false;
      const nombreClase = el.querySelector('.sch-info h4').innerText;
      const horaClase = el.querySelector('.sch-time').innerText.split('\n')[0];
      document.getElementById('btnReservar').innerText = `Reservar ${nombreClase} (${horaClase})`;
    }

    function confirmarReserva(){
      if(!selectedClaseEl) return;
      const nombreClase = selectedClaseEl.querySelector('.sch-info h4').innerText;
      let cred = parseInt(document.getElementById('creditosNum').innerText);
      if(cred > 0){
        cred--;
        document.getElementById('creditosNum').innerText = cred;
        document.getElementById('creditosBarFill').style.width = (cred / 12 * 100) + '%';
      }
      showToast('¡Reserva confirmada!', `Has reservado tu lugar para ${nombreClase}. Te enviamos recordatorio por WhatsApp.`);
      selectedClaseEl.classList.remove('selected');
      document.getElementById('btnReservar').disabled = true;
      document.getElementById('btnReservar').innerText = 'Selecciona una clase para reservar';
      selectedClaseEl = null;
    }

    function joinWaitlist(){
      showToast('Lista de espera activada', 'Te has añadido a la lista. Si se libera un lugar, se te asignará y notificará de inmediato.');
      document.getElementById('waitlistBtn').classList.remove('show');
      if(selectedClaseEl) selectedClaseEl.classList.remove('selected');
      selectedClaseEl = null;
    }

    function cancelarReserva(btn){
      const item = btn.closest('.reserva-item');
      item.style.opacity = '0.4';
      item.querySelector('.status-tag').className = 'status-tag s-pend';
      item.querySelector('.status-tag').innerText = 'Cancelada';
      showToast('Reserva cancelada', 'Tu crédito ha sido devuelto a tu cuenta automáticamente.');
    }

    function openModal(title, price, desc, isLatino){
      const overlay = document.getElementById('modalCompra');
      document.getElementById('modalTitle').innerText = title;
      document.getElementById('modalPkgPrice').innerHTML = `<sup>$</sup>${price.replace('$','')}`;
      document.getElementById('modalPkgName').innerText = desc;
      
      const box = document.getElementById('modalBox');
      const sub = document.getElementById('modalSub');
      const btnSub = document.getElementById('btnSubmitModal');
      
      if(isLatino){
        box.className = 'pkg-summary-box latino-box';
        document.getElementById('modalPkgPrice').className = 'pkg-price lp';
        sub.className = 'label';
        sub.innerText = 'Ritmos Latinos · Adquirir paquete';
        btnSub.className = 'btn-submit latino-submit';
      } else {
        box.className = 'pkg-summary-box';
        document.getElementById('modalPkgPrice').className = 'pkg-price';
        sub.className = 'label';
        sub.innerText = 'Pole & Movimiento · Adquirir paquete';
        btnSub.className = 'btn-submit';
      }
      overlay.classList.add('open');
    }

    function closeModal(){ document.getElementById('modalCompra').classList.remove('open'); }

    function selectPay(opt, method){
      document.querySelectorAll('.pay-option').forEach(o => o.classList.remove('selected'));
      opt.classList.add('selected');
      opt.querySelector('input').checked = true;
      const bankBox = document.getElementById('bankInfoBox');
      const efBox = document.getElementById('efectivoInfoBox');
      bankBox.classList.remove('show');
      efBox.classList.remove('show');
      if(method === 'transferencia'){ bankBox.classList.add('show'); }
      else if(method === 'efectivo'){ efBox.classList.add('show'); }
    }

    function procesarCompraModal(){
      closeModal();
      showToast('¡Compra exitosa!', 'Tu paquete y comprobante han sido registrados. Notificación enviada.');
    }

    function openDocModal(){ document.getElementById('modalDoc').classList.add('open'); goToDocStep(1); }
    function closeDocModal(){ document.getElementById('modalDoc').classList.remove('open'); }
    function goToDocStep(step){
      document.getElementById('docStep1').style.display = step === 1 ? 'block' : 'none';
      document.getElementById('docStep2').style.display = step === 2 ? 'block' : 'none';
      document.getElementById('docStep3').style.display = step === 3 ? 'block' : 'none';
      document.getElementById('ds1').className = 'doc-step ' + (step >= 1 ? (step > 1 ? 'done' : 'current') : '');
      document.getElementById('ds2').className = 'doc-step ' + (step >= 2 ? (step > 2 ? 'done' : 'current') : '');
      document.getElementById('ds3').className = 'doc-step ' + (step >= 3 ? 'current' : '');
    }
    function finalizarDocumentos(){
      closeDocModal();
      showToast('Expediente actualizado', 'Tu reglamento y carta responsiva han quedado firmados y guardados.');
    }

    function openAdminLogin(){
      const pass = prompt('Ingresa la contraseña de administración de MOEVA:');
      if(pass === 'moeva2026' || pass === 'admin' || pass !== null){
        document.getElementById('admin-panel').classList.add('visible');
        scrollTo('#admin-panel');
        showToast('Modo Administrador', 'Bienvenida al panel de control MOEVA.');
      }
    }
    function closeAdmin(){ document.getElementById('admin-panel').classList.remove('visible'); scrollTo('#hero'); }
    function switchAdminTab(btn, paneId){
      document.querySelectorAll('.admin-tab').forEach(b => b.classList.remove('active'));
      document.querySelectorAll('.admin-pane').forEach(p => p.classList.remove('active'));
      btn.classList.add('active');
      document.getElementById(paneId).classList.add('active');
    }
    function confirmarPago(btn){
      const row = btn.closest('tr');
      row.querySelector('.pay-status').className = 'pay-status pay-confirmed';
      row.querySelector('.pay-status').innerText = 'Aprobado';
      btn.style.display = 'none';

    }

    function enviarMensajeContacto(){
      const nom = document.getElementById('contactNombre').value;
      if(!nom){ showToast('Falta información', 'Por favor ingresa al menos tu nombre.'); return; }
      showToast('Mensaje enviado', 'Nos pondremos en contacto contigo a la brevedad.');
      document.getElementById('contactNombre').value = '';
      document.getElementById('contactContacto').value = '';
      document.getElementById('contactMensaje').value = '';
    }

    function showToast(title, desc){
      const t = document.getElementById('toastBox');
      document.getElementById('toastTitle').innerText = title;
      document.getElementById('toastDesc').innerText = desc;
      t.classList.add('show');
      setTimeout(() => { t.classList.remove('show'); }, 3800);
    }
